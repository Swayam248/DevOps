# sudo su -

root# yum install wget -y

//go to Jenkins in AWS page...and paste below lines

root# rpm --import https://pkg.jenkins.io/redhat-stable/jenkins.io-2023.key
root# yum upgrade
root# yum install java-21-amazon-corretto -y
root# yum install jenkins -y
root# systemctl enable Jenkins
root# systemctl start Jenkins
